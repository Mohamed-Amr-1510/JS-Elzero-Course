```js
/*
  Type Coercion (Type Casting)
  - +
  - -
  - "" - 2
  - false - true
*/

let a = "100";
let b = 20;
let c = true;

console.log(+a + b + c);       //121
console.log(a + b );           //10020
console.log( a - b );          //80
console.log( "" - 2 );         //-2
console.log( false - true );   //-1
console.log(a + b + true );    //10020true
```
